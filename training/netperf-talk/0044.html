The following batch file under NT does the trick quite well:

start/b netperf -p 12865 -H 192.159.80.11 -- -s %1 -S %1 >run.1
start/b/w netperf -p 12866 -H 192.159.80.11 -- -s %1 -S %1 >run.2
pause Press enter to view results
type run.1 run.2
